18 #define IS_S_TYPE(x) (t[x])
19 #define IS_L_TYPE(x) (!t[x])
20 #define IS_LMS_TYPE(x) (IS_S_TYPE(x) && x != 0 && IS_L_TYPE(x - 1))
21  t.back() = true;
22  for (int i = n - 2; i >= 0; --i) {
23  t[i] = (s[i] < s[i + 1] || (IS_S_TYPE(i + 1) && s[i] == s[i + 1]));
24  if (IS_LMS_TYPE(i + 1)) p1.push_back(i + 1);
25  }
